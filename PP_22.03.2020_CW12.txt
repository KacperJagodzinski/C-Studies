#include <iostream>
#include <cstdio>
using namespace std;

int main(){
  //Napisaæ program, który liczy sumê, max, min i œredni¹ wartoœci z tablicy dwuwymiarowej – 9 elementów.
  int tab[3][3];
  for(int i=0; i<3;i++){
    for(int j=0;j<3;j++){
      cout<<"WprowadŸ element tablicy: ";
      cin>>tab[i][j];
    }
  }
  int max=tab[0][0],min = tab[0][0],suma=0,srednia;
  for(int i=0; i<3;i++){
    for(int j=0;j<3;j++){
      if(tab[i][j]>max){
        max = tab[i][j];
      }
      if(tab[i][j]<min){
        min = tab[i][j];
      }
      suma+=tab[i][j];
    }
  }
  srednia = suma / 9;
  cout<<"MAX = " << max<<endl<<"MIN = " << min <<endl<< "Œrednia = " << srednia;
  
  return 0;
}
