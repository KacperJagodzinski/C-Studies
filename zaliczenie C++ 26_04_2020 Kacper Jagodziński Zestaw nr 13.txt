int silnia(int n)
{
  int silniaWynik=1;
  for (int i = 1; i <= n; i++){
    silniaWynik *= i;
  }
  return silniaWynik;
}

bool zadanieA(double a, int n, int x)
{
  int nSilnia;
  bool czyPrawidlowe = true;
  //obliczam silnie za pomoca zewnetrznej metody
  nSilnia = silnia(n);
  //sprawdzam czy mianownik jest rowny 0 
  if ((((a * x * x) - 2) * nSilnia) == 0 ){
    czyPrawidlowe = false;
    return czyPrawidlowe;
  }else{
    return czyPrawidlowe;
  }
}
